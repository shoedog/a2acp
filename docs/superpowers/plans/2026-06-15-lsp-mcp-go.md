  ```rust
  /// Resolve a bare server program name to an absolute path. Search order:
  /// 1. name already containing a path separator → returned unchanged;
  /// 2. each `$PATH` dir; 3. `$HOME/.local/bin`; 4. `$HOME/.cargo/bin`;
  /// 5. each dir in `go_candidates` (e.g. `$GOPATH/bin`, `$(go env GOROOT)/bin`) — for off-PATH gopls.
  /// First `is_usable_interpreter`-passing candidate wins; else the bare `name` (caller degrades to the OS
  /// "not found" error). Factored to accept explicit `path_var`/`home_dir`/`go_candidates` so tests are hermetic.
  fn resolve_lsp_server_with_env(
      name: &str,
      path_var: Option<&str>,
      home_dir: Option<&str>,
      go_candidates: &[std::path::PathBuf],
  ) -> String {
      use std::path::MAIN_SEPARATOR;
      if name.contains(MAIN_SEPARATOR) {
          return name.to_string();
      }
      if let Some(path) = path_var {
          for dir in path.split(':').filter(|d| !d.is_empty()) {
              let candidate = std::path::Path::new(dir).join(name);
              if is_usable_interpreter(&candidate) {
                  return candidate.to_string_lossy().into_owned();
              }
          }
      }
      if let Some(home) = home_dir {
          let home = std::path::Path::new(home);
          for subdir in &[".local/bin", ".cargo/bin"] {
              let candidate = home.join(subdir).join(name);
              if is_usable_interpreter(&candidate) {
                  return candidate.to_string_lossy().into_owned();
              }
          }
      }
      // Go toolchain locations (for a bare `gopls` not on PATH, e.g. in a container).
      for dir in go_candidates {
          let candidate = dir.join(name);
          if is_usable_interpreter(&candidate) {
              return candidate.to_string_lossy().into_owned();
          }
      }
      name.to_string()
  }

  /// Best-effort Go toolchain bin dirs from `go env` (`$GOPATH/bin`, `$(go env GOROOT)/bin`). Skipped (empty)
  /// if `go` is absent — never panics. Used only to find a bare `gopls` that isn't on PATH.
  fn go_bin_candidates() -> Vec<std::path::PathBuf> {
      let mut out = Vec::new();
      if let Ok(o) = std::process::Command::new("go").args(["env", "GOPATH"]).output() {
          if o.status.success() {
              let p = String::from_utf8_lossy(&o.stdout).trim().to_string();
              if !p.is_empty() {
                  out.push(std::path::PathBuf::from(p).join("bin"));
              }
          }
      }
      if let Ok(o) = std::process::Command::new("go").args(["env", "GOROOT"]).output() {
          if o.status.success() {
              let p = String::from_utf8_lossy(&o.stdout).trim().to_string();
              if !p.is_empty() {
                  out.push(std::path::PathBuf::from(p).join("bin"));
              }
          }
      }
      out
  }
  ```
