use std::time::Instant;
/// Injected time source — keeps bridge-core's no-`Date::now` discipline at the lib boundary
/// (the binary supplies the real clock). Covers BOTH monotonic (idle TTL) and wall-clock (task rows).
pub trait Clock: Send + Sync {
    fn now_instant(&self) -> Instant;
    fn now_ms(&self) -> i64;
}
pub struct SystemClock;
impl Clock for SystemClock {
    fn now_instant(&self) -> Instant { Instant::now() }
    fn now_ms(&self) -> i64 {
        std::time::SystemTime::now().duration_since(std::time::UNIX_EPOCH)
            .map(|d| d.as_millis() as i64).unwrap_or(0)
    }
}

#[cfg(any(test, feature = "test-clock"))]
impl Clock for FixedClock {
    fn now_instant(&self) -> Instant { self.base }
